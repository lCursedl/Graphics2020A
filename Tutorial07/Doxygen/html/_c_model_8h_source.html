100  CMesh processMesh(aiMesh *mesh, const aiScene *scene)
101  {
102  // data to fill
103  std::vector<Vertex> vertices;
104  std::vector<unsigned int> indices;
105  std::vector<Texture> textures;
106 
107  // Walk through each of the mesh's vertices
108  for (unsigned int i = 0; i < mesh->mNumVertices; i++)
109  {
110  Vertex vertex;
111  glm::vec3 vector; // we declare a placeholder vector since assimp uses its own vector class that doesn't directly convert to glm's vec3 class so we transfer the data to this placeholder glm::vec3 first.
112  // positions
113  vector.x = mesh->mVertices[i].x;
114  vector.y = mesh->mVertices[i].y;
115  vector.z = mesh->mVertices[i].z;
116  vertex.Position = vector;
117  // normals
118  vector.x = mesh->mNormals[i].x;
119  vector.y = mesh->mNormals[i].y;
120  vector.z = mesh->mNormals[i].z;
121  vertex.Normal = vector;
122  // texture coordinates
123  if (mesh->mTextureCoords[0]) // does the mesh contain texture coordinates?
124  {
125  glm::vec2 vec;
126  // a vertex can contain up to 8 different texture coordinates. We thus make the assumption that we won't
127  // use models where a vertex can have multiple texture coordinates so we always take the first set (0).
128  vec.x = mesh->mTextureCoords[0][i].x;
129  vec.y = mesh->mTextureCoords[0][i].y;
130  vertex.TexCoords = vec;
131  }
132  else
133  vertex.TexCoords = glm::vec2(0.0f, 0.0f);
134  // tangent
135  vector.x = mesh->mTangents[i].x;
136  vector.y = mesh->mTangents[i].y;
137  vector.z = mesh->mTangents[i].z;
138  vertex.Tangent = vector;
139  // bitangent
140  vector.x = mesh->mBitangents[i].x;
141  vector.y = mesh->mBitangents[i].y;
142  vector.z = mesh->mBitangents[i].z;
143  vertex.Bitangent = vector;
144  vertices.push_back(vertex);
145  }
146  // now wak through each of the mesh's faces (a face is a mesh its triangle) and retrieve the corresponding vertex indices.
147  for (unsigned int i = 0; i < mesh->mNumFaces; i++)
148  {
149  aiFace face = mesh->mFaces[i];
150  // retrieve all indices of the face and store them in the indices vector
151  for (unsigned int j = 0; j < face.mNumIndices; j++)
152  indices.push_back(face.mIndices[j]);
153  }
154  // process materials
155  aiMaterial* material = scene->mMaterials[mesh->mMaterialIndex];
156  // we assume a convention for sampler names in the shaders. Each diffuse texture should be named
157  // as 'texture_diffuseN' where N is a sequential number ranging from 1 to MAX_SAMPLER_NUMBER.
158  // Same applies to other texture as the following list summarizes:
159  // diffuse: texture_diffuseN
160  // specular: texture_specularN
161  // normal: texture_normalN
162 
163  // 1. diffuse maps
164  std::vector<Texture> diffuseMaps = loadMaterialTextures(material, aiTextureType_DIFFUSE, "texture_diffuse");
165  textures.insert(textures.end(), diffuseMaps.begin(), diffuseMaps.end());
166  // 2. specular maps
167  std::vector<Texture> specularMaps = loadMaterialTextures(material, aiTextureType_SPECULAR, "texture_specular");
168  textures.insert(textures.end(), specularMaps.begin(), specularMaps.end());
169  // 3. normal maps
170  std::vector<Texture> normalMaps = loadMaterialTextures(material, aiTextureType_HEIGHT, "texture_normal");
171  textures.insert(textures.end(), normalMaps.begin(), normalMaps.end());
172  // 4. height maps
173  std::vector<Texture> heightMaps = loadMaterialTextures(material, aiTextureType_AMBIENT, "texture_height");
174  textures.insert(textures.end(), heightMaps.begin(), heightMaps.end());
175 
176  // return a mesh object created from the extracted mesh data
177  return CMesh(vertices, indices, textures);
178  }
185  std::vector<Texture> loadMaterialTextures(aiMaterial *mat, aiTextureType type, std::string typeName)
186  {
187  std::vector<Texture> textures;
188  for (unsigned int i = 0; i < mat->GetTextureCount(type); i++)
189  {
190  aiString str;
191  mat->GetTexture(type, i, &str);
192  // check if texture was loaded before and if so, continue to next iteration: skip loading a new texture
193  bool skip = false;
194  for (unsigned int j = 0; j < textures_loaded.size(); j++)
195  {
196  if (std::strcmp(textures_loaded[j].path.data(), str.C_Str()) == 0)
197  {
198  textures.push_back(textures_loaded[j]);
199  skip = true; // a texture with the same filepath has already been loaded, continue to next one. (optimization)
200  break;
201  }
202  }
203  if (!skip)
204  { // if texture hasn't been loaded already, load it
205  Texture texture;
206  texture.id = TextureFromFile(str.C_Str(), this->directory);
207  texture.type = typeName;
208  texture.path = str.C_Str();
209  textures.push_back(texture);
210  textures_loaded.push_back(texture); // store it as texture loaded for entire model, to ensure we won't unnecesery load duplicate textures.
211  }
212  }
